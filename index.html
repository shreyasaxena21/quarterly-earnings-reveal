Code Sample â€” Simple NPV (Python)


import math
from typing import Iterable

def npv(rate: float, cashflows: Iterable[float]) -> float:
    """Net Present Value excluding C0 (use negative initial investment in cashflows if needed)."""
    return sum(cf / ((1 + rate) ** t) for t, cf in enumerate(cashflows, start=1))

# Example: 10% discount rate, yearly inflows
print(npv(0.10, [100, 120, 140, 160]))
